//Decoder code of VHDL
library IEEE;
use IEEE.std_logic_1164.all;

entity DEcoder is
	port(A:in std_logic;
		B:in std_logic;
		D0:out std_logic;
		D1:out std_logic;
		D2:out std_logic;
		D3:out std_logic);
end DEcoder;
architecture DecoderLogic of Decoder is 
begin 
	D0<=(NOT A) AND (NOT B);
	D1<=(NOT A) AND B;
	D2<=A AND (NOT B);
	D3<=A AND B;
end DecoderLogic;
